149 FolderID id,
150 KnownFolderFlag flags,
151 void* user_handle
152 ) noexcept(false) -> fs::path
153 {
154 auto buf = PWSTR();
155 auto const res = SHGetKnownFolderPath(
156 ::folder_id_to_winapi_folder_id(id),
157 static_cast<DWORD>(flags),
158 user_handle,
159 &buf
160 );
161 if(res != S_OK) {
162 CoTaskMemFree(buf);
163 throw winapi::windows_exception("SHGetKnownFolderPath returned error");
164 }
165 auto const path = fs::path(buf);
166 CoTaskMemFree(buf);
167 return path;
168 }
